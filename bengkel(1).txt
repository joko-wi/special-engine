    function initialize() { 
        var map;
        var bounds = new google.maps.LatLngBounds();
        var mapOptions = {
            mapTypeControl: false,
            mapTypeControlOptions: {
                mapTypeIds: [google.maps.MapTypeId.ROADMAP, 'map_style']
            },
            scrollwheel: false,
            zoom: 17,
            draggable: true
        };
        map = new google.maps.Map(document.getElementById("map_canvas"), mapOptions);
        map.setTilt(45);
        
        // Multiple Markers
        var markers = [
        [ 'PRATAMA MOTOR', -6.290728, 107.1513634]
        ];
        
        // Info Window Content
        var infoWindowContent = [
        ['<div class="info_content">' +
        '<h4>PRATAMA MOTOR</h4>' +
        '<p>Jl Industri Cikarang Pasir Gombong No.88B, Pasir Gombong, Cikarang Utara, Bekasi, Jawa Barat 17530 </p>' + '</div>']
        ];
        
        // Display multiple markers on a map
        var infoWindow = new google.maps.InfoWindow(), marker, i;
        
        // Loop through our array of markers & place each one on the map  
        for( i = 0; i < markers.length; i++ ) {
            var position = new google.maps.LatLng(markers[i][1], markers[i][2]);
            bounds.extend(position);
            marker = new google.maps.Marker({
                position: position,
                map: map,
                title: markers[i][0]
            });
            
            // Allow each marker to have an info window    
            google.maps.event.addListener(marker, 'click', (function(marker, i) {
                return function() {
                    infoWindow.setContent(infoWindowContent[i][0]);
                    infoWindow.open(map, marker);
                }
            })(marker, i));

            // Automatically center the map fitting all markers on the screen
            map.fitBounds(bounds);
        }

        // Override our map zoom level once our fitBounds function runs (Make sure it only runs once)
        var boundsListener = google.maps.event.addListener((map), 'bounds_changed', function(event) {
            this.setZoom(17);
            google.maps.event.removeListener(boundsListener);
            
        });
    }

function getRasio(){
    var diameter = $('#diameter').val();
    $.ajax({
        url: 'https://www.tomonet.co.id/web/home/getRasio',
        dataType: 'json', type: 'post', data: "diameter="+diameter,
        success: function(data){
            $('#ratio, #velg').empty();
            $('#ratio, #velg').append($("<option></option>").attr("value", '').text('-- Pilih Rasio --'));
            $.each(data.filter_aspect_ratio,function(key,value){
                $('#ratio').append($("<option></option>").attr("value", value.title).text(value.title));
            });
            return false;
        }
    });

}
